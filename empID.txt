#include <stdio.h>
#include <stdlib.h>

int main() {
    int *empID;
    int n, extra, i;

    
    printf("Enter number of employees: ");
    scanf("%d", &n);


    empID = (int *)malloc(n * sizeof(int));

    if (empID == NULL) {
        printf("Memory allocation failed\n");
        return 0;
    }

    
    printf("Enter employee IDs:\n");
    for (i = 0; i < n; i++) {
        scanf("%d", &empID[i]);
    }

    
    printf("Enter number of additional employees: ");
    scanf("%d", &extra);

    empID = (int *)realloc(empID, (n + extra) * sizeof(int));

    if (empID == NULL) {
        printf("Memory reallocation failed\n");
        return 0;
    }

    
    printf("Enter new employee IDs:\n");
    for (i = n; i < n + extra; i++) {
        scanf("%d", &empID[i]);
    }

    
    printf("Employee IDs list:\n");
    for (i = 0; i < n + extra; i++) {
        printf("%d ", empID[i]);
    }

    
    free(empID);

    return 0;
}
