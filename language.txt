#include <stdio.h>

int main() {
    int choice;
    printf("Welcome to IVR System\n");
    printf("Press 1 for Tamil\n");
    printf("Press 2 for English\n");
    printf("Press 3 for Hindi\n");
    printf("Enter your choice:\n");
    scanf("%d", &choice);
    switch(choice){
        case 1:
        printf("You have chosen Tamil\n");
        break;
        case 2:
        printf("You have chosen English\n");
        break;
        case 3:
        printf("You have chosen Hindi\n");
        break;
        default:
        printf("Invalid choice\n");
        
    }
        

    return 0;
}
